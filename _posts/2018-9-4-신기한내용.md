            var sayHello = function (name) {
              var text = 'Hello, ' + name;
              return function () {
                console.log(text);
              };
            };// 클로저 생성.

            var helloTodd = sayHello('Todd'); 
            helloTodd(); //이것은 closure를 호출하여 'Hello, Todd'를 출력한다.
